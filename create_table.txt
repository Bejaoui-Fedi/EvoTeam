CREATE TABLE `exercise` (
  `id_exercise` bigint(20) unsigned NOT NULL AUTO_INCREMENT,
  `user_id` int(11) NOT NULL,
  `objectiveId` bigint(20) unsigned NOT NULL,
  `title` varchar(150) NOT NULL,
  `description` text NOT NULL,
  `type` enum('respiration','journaling','meditation','cbt','challenge','relaxation') NOT NULL,
  `durationMinutes` smallint(5) unsigned NOT NULL,
  `difficulty` enum('debutant','moyen','avance') NOT NULL,
  `mediaUrl` varchar(255) DEFAULT NULL,
  `steps` longtext CHARACTER SET utf8mb4 COLLATE utf8mb4_bin DEFAULT NULL CHECK (json_valid(`steps`)),
  `orderIndex` int(10) unsigned DEFAULT NULL,
  `isPublished` tinyint(1) NOT NULL DEFAULT 0,
  `createdAt` timestamp NOT NULL DEFAULT current_timestamp(),
  `updatedAt` timestamp NOT NULL DEFAULT current_timestamp() ON UPDATE current_timestamp(),
  PRIMARY KEY (`id_exercise`),
  KEY `idx_objective` (`objectiveId`),
  KEY `idx_published` (`isPublished`),
  KEY `idx_type` (`type`),
  KEY `idx_difficulty` (`difficulty`),
  KEY `idx_exercise_user` (`user_id`),
  CONSTRAINT `fk_exercise_objective` FOREIGN KEY (`objectiveId`) REFERENCES `objective` (`id_objective`) ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT `fk_exercise_user` FOREIGN KEY (`user_id`) REFERENCES `user` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=7 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
